% \art[20.416]{2}→ «artículo 2 de la Ley 20.416»
\newcommand{\art}[2][20.720]{%
  artículo~#2\ifthenelse{\equal{#1}{20.720}}{}{ de la Ley~\Nro#1}%

\newcommand{\Art}[2][20.720]{%
  Artículo~#2\ifthenelse{\equal{#1}{20.720}}{}{ de la Ley~\Nro#1}%
  \index[leyes]{Ley #1@Ley~\Nro#1!art. #2}}
